## setup.sh

Path: `scripts/setup.sh`

One-command project setup for new developers.

```bash
#!/bin/bash
# Project setup script for new developers.
# Usage: bash scripts/setup.sh

set -e

echo "=== Project Setup ==="

# Check prerequisites
command -v git >/dev/null 2>&1 || { echo "ERROR: git is required"; exit 1; }

# Detect project type and install dependencies
if [ -f "package.json" ]; then
    echo "Installing Node.js dependencies..."
    if command -v pnpm >/dev/null 2>&1; then
        pnpm install
    elif command -v yarn >/dev/null 2>&1; then
        yarn install
    else
        npm install
    fi
elif [ -f "pyproject.toml" ]; then
    echo "Installing Python dependencies..."
    python3 -m venv .venv
    source .venv/bin/activate
    pip install -e ".[dev]"
elif [ -f "go.mod" ]; then
    echo "Installing Go dependencies..."
    go mod download
elif [ -f "Cargo.toml" ]; then
    echo "Installing Rust dependencies..."
    cargo build
fi

# Setup environment
if [ -f ".env.example" ] && [ ! -f ".env" ]; then
    echo "Creating .env from .env.example..."
    cp .env.example .env
    echo "IMPORTANT: Edit .env with your actual values"
fi

# Setup Git hooks
if [ -f ".claude/hooks/check-doc-sync.sh" ]; then
    chmod +x .claude/hooks/*.sh
    echo "Claude hooks configured"
fi

if [ -d ".git" ]; then
    # Install commit-msg hook
    if [ -f "scripts/install-hooks.sh" ]; then
        bash scripts/install-hooks.sh
    fi
fi

echo "=== Setup Complete ==="
echo "Next steps:"
echo "  1. Edit .env with your configuration"
echo "  2. Read CLAUDE.md for project conventions"
echo "  3. Read docs/onboarding.md for development workflow"
```

## deploy.sh

Path: `scripts/deploy.sh`

Deployment script with pre-checks and verification.

```bash
#!/bin/bash
# Deploy to target environment.
# Usage: bash scripts/deploy.sh [environment]
# Environments: staging, production (default: staging)

set -e

ENV="${1:-staging}"
echo "=== Deploying to $ENV ==="

# Pre-deploy checks
echo "Running pre-deploy checks..."

# Check clean working tree
if [ -n "$(git status --porcelain)" ]; then
    echo "ERROR: Working tree is not clean. Commit or stash changes first."
    exit 1
fi

# Check branch
BRANCH=$(git branch --show-current)
if [ "$ENV" = "production" ] && [ "$BRANCH" != "main" ]; then
    echo "WARNING: Deploying to production from branch '$BRANCH' (not main)"
    read -p "Continue? (y/N) " -n 1 -r
    echo
    [[ ! $REPLY =~ ^[Yy]$ ]] && exit 1
fi

# Run tests
echo "Running tests..."
# <INSERT_TEST_COMMAND>

# Build
echo "Building..."
# <INSERT_BUILD_COMMAND>

# Deploy
echo "Deploying to $ENV..."
# <INSERT_DEPLOY_COMMAND>

# Verify
echo "Verifying deployment..."
# <INSERT_VERIFY_COMMAND>

echo "=== Deployment to $ENV complete ==="
```

## install-hooks.sh

Path: `scripts/install-hooks.sh`

Install Git hooks that are not managed by Claude settings.

```bash
#!/bin/bash
# Install Git hooks.
# Usage: bash scripts/install-hooks.sh

set -e

HOOKS_DIR=".git/hooks"

if [ ! -d "$HOOKS_DIR" ]; then
    echo "ERROR: .git/hooks directory not found. Is this a git repository?"
    exit 1
fi

# Install commit-msg hook (removes Co-Authored-By lines)
cat > "$HOOKS_DIR/commit-msg" << 'HOOK'
#!/bin/bash
# Remove Co-Authored-By lines from commit messages.
# Prevents Claude and other AI assistants from appearing as contributors.
sed -i '/^[Cc]o-[Aa]uthored-[Bb]y:.*/d' "$1"
sed -i -e :a -e '/^\n*$/{$d;N;ba' -e '}' "$1"
HOOK
chmod +x "$HOOKS_DIR/commit-msg"
echo "Installed commit-msg hook (AI co-author removal)"

echo "=== Git hooks installed ==="
```
